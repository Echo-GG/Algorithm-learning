## 算法思想

双指针法(**滑动窗口**)

## 思路分析:

定义 left 和 right 指针，先向右移动 right 指针扩大窗口，直至寻找到完全覆盖子串的字符串区间，此后向右移动 left 指针，尝试在不影响结果正确性的前提下，缩小窗口，从而找到最小的字符串覆盖区间。

## AC代码如下:
```C++
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    string minWindow(string s, string t) {
        unordered_map<char,int> window,need;
        int n1 = s.size();
        int n2 = t.size();
        for(char ch:t)
        {
            need[ch]++;
        }

        int left = 0;
        int right = 0;
        int count = 0;
        int min_size = INT_MAX;
        int min_left = 0;

        while(right < n1)
        {
            char ch1 = s[right];
            right++;
            if(need.count(ch1))
            {
                window[ch1]++;
                if(window[ch1] == need[ch1])
                {
                    count++;
                }
            }
        

            while(count == need.size())
            {
                if(right - left < min_size)
                {
                    min_size = right - left;
                    min_left = left;
                }

                char ch2 = s[left];
                left++;
                if(need.count(ch2))
                {
                    if(window[ch2] == need[ch2])
                    {
                        count--;
                    }

                    window[ch2]--;
                }

            }
        }
        return min_size==INT_MAX?"":s.substr(min_left,min_size);
    }
};
```

### 使用ASCII字符集 vector \<int> chars(128,0), vector \<bool> flag(128,false)，替代 unordered_map 实现算法效率的优化:
```C++
#include <bits/stdc++.h>
using namespace std;
class Solution {
public:
    string minWindow(string s, string t) {
        vector<bool> flag(128,false);
        vector<int> chars(128,0);
        for(int i=0;i<t.size();i++)
        {
            flag[t[i]] = true;
            ++chars[t[i]];
        }
        int cnt = 0;
        int l = 0;
        int min_size = INT_MAX;
        int min_l = 0;
        for(int r=0;r<s.size();r++)
        {
            // 扩张窗口
            if(flag[s[r]])
            {
                chars[s[r]]--;
                if(chars[s[r]]>=0)
                {
                    cnt++;
                }
            }

            while(cnt==t.size())
            {
                if(r - l + 1< min_size)
                {
                    min_l = l;
                    min_size = r - l + 1;
                }

                // 在正确的前提下，缩小窗口

                // 当前被检查的字符是所需要的且不被再需要，如果左移 left 则会失去他：
                // 等价于种类数 - 1， 对应的被需要的字符 + 1
                if(flag[s[l]] && chars[s[l]]==0)
                {
                    --cnt;
                }
                chars[s[l]]++;

                ++l;
            }
        }
        return min_size > s.size() ? "" : s.substr(min_l,min_size);
    }
};


```
